   \*\*
   \*\*
  \*\*\*\*
  \*\*\*\*
 \*\*\*\*\*\*

where number of pairs line is given by user
*/
#include <iostream>

/** main function */
int main() {
    int l, st = 2, x, r, z, n, sp;
    std::cout << "Enter number of pair - ";
    std::cin >> x;
    z = x;
    for (r = 1; r <= x; r++) {
        z = z - 1;
        for (n = 1; n <= 2; n++) {
            for (sp = 1; sp <= z; sp++) {
                std::cout << " ";
            }
            for (l = 1; l <= st; l++) {
                std::cout << "\\*";
            }
            std::cout << std::endl;
        }
        st = st + 2;
    }
}

The given code uses two nested loops (one with the outer loop and one with the inner loop) which runs n times (where n is the number of pairs of lines). The time it takes to run the code increases quadratically with the increasing value of n. Therefore, the given code is said to have a time complexity of O(n^2).</s>
