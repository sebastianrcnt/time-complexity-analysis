public class Main {
  public static void main(String args[]) {
    Scanner sc = new Scanner(System.in);

    int m = sc.nextInt();

    long totalBlocks = 0;

      a[i] = sc.nextLong();
      totalBlocks += a[i];
    }

    Arrays.sort(a);

      if (a[i] > selected) selected++;
    }

    long remBlocks = totalBlocks - leftCols - n;

    System.out.print(remBlocks);
  }
