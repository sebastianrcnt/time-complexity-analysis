# returns (F(n), F(n-1))
def _fib(n: int) -> tuple[int, int]:
    if n == 0:  # (F(0), F(1))
        return (0, 1)

    # F(2n) = F(n)[2F(n+1) âˆ’ F(n)]
    # F(2n+1) = F(n+1)^2+F(n)^2

    d = a * a + b * b
    return (d, c + d) if n % 2 else (c, d)
