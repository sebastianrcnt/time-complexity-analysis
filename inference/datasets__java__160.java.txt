    if (right == null) return true;

    boolean isp = isPalindromeUtil(right.next);

    if (isp == false) return false;

    boolean isp1 = (right.data == (left).data);

    left = left.next;

    return isp1;
  }

    boolean result = isPalindromeUtil(head);

    return result;
